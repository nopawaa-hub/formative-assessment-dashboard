function doGet(e) {
  var action = e.parameter.action;
  var ss = SpreadsheetApp.openById(e.parameter.sheetId
    || 'YOUR_SHEET_ID');
  var sheet = ss.getSheets()[0];
  var data = sheet.getDataRange().getValues();
  var headers = data[0];
  var responses = data.slice(1).map(function(row) {
    var obj = {};
    headers.forEach(function(h,i){ obj[h] = row[i]; });
    return {
      name: obj['Nama'] || obj['Name'] || '',
      class: obj['Kelas'] || obj['Class'] || '',
      readingScore: obj['Score'] || obj['Skor'] || null
    };
  });
  return ContentService
    .createTextOutput(JSON.stringify({responses:responses}))
    .setMimeType(ContentService.MimeType.JSON);
}
